q:is there any object with color of blue
p:
        (exists
            (intersect 
                (forall
                    (union
                        (Pr (color (expand (scene $0)) ) blue)
                        (not(expand (scene $0))) 
                    )
                )
                (scene $0)         
            )
        )
        
answer:yes
